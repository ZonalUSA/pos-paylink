// Pay by Link tip API (JSON). // // The customer-facing tip *page* is a static HTML file hosted in Supabase // Storage (the Edge Functions gateway rewrites text/html → text/plain on the // default domain, so HTML can't be served from here). That page calls this // function for data and to create the payment link. // // Endpoints (same origin as the Storage page, so no CORS issues): // GET ?order= // → { ok, orderNumber, amount, tips:[{pct,amount}] } or { ok:false, reason } // GET ?order=&action=create&tip= // → { ok, url } or { ok:false, reason } // // Amount is read from orders.pay_link_amount (authoritative — never supplied by // the client). Adyen credentials + tip presets come from location_settings. // // Deploy: Edge Function `pay-tip`, paste this index.ts, DISABLE "Verify JWT". import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!; const SERVICE_ROLE = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!; const supabase = createClient(SUPABASE_URL, SERVICE_ROLE); const CORS = { "access-control-allow-origin": "*", "access-control-allow-headers": "*", "access-control-allow-methods": "GET, OPTIONS", }; function json(obj: unknown, status = 200): Response { return new Response(JSON.stringify(obj), { status, headers: { "content-type": "application/json", "cache-control": "no-store", ...CORS }, }); } Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: CORS }); const url = new URL(req.url); const orderId = url.searchParams.get("order"); if (!orderId) return json({ ok: false, reason: "missing" }, 400); const { data: order, error } = await supabase .from("orders") .select("order_number, pay_link_amount, paid_at, pay_link_items, pay_link_summary") .eq("id", orderId) .maybeSingle(); if (error) return json({ ok: false, reason: "error" }, 500); if (!order) return json({ ok: false, reason: "notfound" }); if (order.paid_at) return json({ ok: false, reason: "paid" }); const bill = Number(order.pay_link_amount ?? 0); if (!(bill > 0)) return json({ ok: false, reason: "nothing" }); // select("*") so a not-yet-added column never fails the whole read (missing // columns simply come back undefined and fall back to defaults below). const { data: s, error: sErr } = await supabase .from("location_settings") .select("*") .eq("id", "default") .maybeSingle(); if (sErr) console.error("location_settings read failed", sErr); // ── Create link ─────────────────────────────────────────────────────────── if (url.searchParams.get("action") === "create") { const apiKey = (s?.adyen_api_key ?? "") as string; const merchant = (s?.adyen_merchant_account ?? "") as string; if (!apiKey || !merchant) { console.error("Missing Adyen creds in location_settings", { hasKey: !!apiKey, hasMerchant: !!merchant, rowFound: !!s, }); return json({ ok: false, reason: "config" }, 500); } const tip = Math.max(0, parseInt(url.searchParams.get("tip") ?? "0", 10)) / 100; const total = bill + tip; // Optional shopper email → Adyen sends its payment confirmation email if // the confirmation-email setting is enabled in the Customer Area. const emailRaw = (url.searchParams.get("email") ?? "").trim(); const shopperEmail = /^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(emailRaw) ? emailRaw : ""; const testEnv = (s?.adyen_test_environment ?? true) as boolean; const prefix = (s?.adyen_live_url_prefix ?? "") as string; const host = testEnv ? "checkout-test.adyen.com" : `${prefix}-checkout-live.adyenpayments.com`; const path = testEnv ? "/v71/paymentLinks" : "/checkout/v71/paymentLinks"; try { const resp = await fetch(`https://${host}${path}`, { method: "POST", headers: { "content-type": "application/json", "x-api-key": apiKey, }, body: JSON.stringify({ merchantAccount: merchant, amount: { currency: "USD", value: Math.round(total * 100) }, reference: `brain-pos-${order.order_number}`, countryCode: "US", ...(shopperEmail ? { shopperEmail } : {}), metadata: { orderId, orderNumber: String(order.order_number), tip: String(Math.round(tip * 100)), }, }), }); if (!resp.ok) { console.error("paymentLinks failed", resp.status, await resp.text()); return json({ ok: false, reason: "adyen" }, 502); } const link = await resp.json(); return json({ ok: true, url: link.url }); } catch (e) { console.error("paymentLinks exception", e); return json({ ok: false, reason: "adyen" }, 502); } } // ── Bill + tip options ────────────────────────────────────────────────────── const pcts = [ Number(s?.adyen_tip_percent_1 ?? 15), Number(s?.adyen_tip_percent_2 ?? 18), Number(s?.adyen_tip_percent_3 ?? 20), ].filter((p) => p > 0); const lines = Array.isArray(order.pay_link_items) ? (order.pay_link_items as Array>).map((i) => ({ name: String(i.name ?? "Item"), qty: Number(i.qty ?? 1), amount: Number(i.amount ?? 0), })) : []; const sum = (order.pay_link_summary ?? {}) as Record; const summary = { subtotal: Number(sum.subtotal ?? 0), discount: Number(sum.discount ?? 0), serviceCharge: Number(sum.serviceCharge ?? 0), tax: Number(sum.tax ?? 0), total: Number(sum.total ?? bill), }; return json({ ok: true, orderNumber: order.order_number, amount: Number(bill.toFixed(2)), tips: pcts.map((p) => ({ pct: p, amount: Number((bill * p / 100).toFixed(2)) })), brand: (s?.business_name ?? "") as string, logo: (s?.logo_url ?? "") as string, lines, summary, }); });
